

import javafx.application.Application;
import javafx.event.ActionEvent;
import javafx.event.EventHandler;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.layout.StackPane;
import javafx.stage.Stage;



public class HelloWorld extends Application {
    
    @Override
    /**
     * provide the implementation for start.  This is where 
     * program execution begins. Used to initialize the GUI.
     * 
     */
    public void start(Stage primaryStage) {
    	
    	// create GUI control
        Button btn = new Button();
        
        // set properties
        btn.setText("Say 'Hello World'");
        
        // set event handler
        btn.setOnAction(
        		
        		
        		/* This is an anonymous object...it has no named type
        		 * so it can not be referred to outside of this context.
        		 * We just know that the object implements the EventHandler
        		 * interface.
        		 */ 
        		new EventHandler<ActionEvent>() {
 
        			@Override
        			public void handle(ActionEvent event) {
        				System.out.println("Hello World!");
        			}	
        		}
        		
        );
        
        
        /* a pane retains a set of references to 
         * controls and dictates their positions on the screen. 
         */
        StackPane root = new StackPane();
        root.getChildren().add(btn);

        /* scene is a GUI control, that acts as a container for 
         * other controls.
         */
        Scene scene = new Scene(root, 300, 250);

        primaryStage.setTitle("Hello World!");
        primaryStage.setScene(scene);
        primaryStage.show();
    }
    
    /**
     * 
     * Program is started here.
     * @param args
     */
    public static void main(String[] args) {
    	// initialization process is spawned
        Application.launch(args);
    }
}
